program 
  var N : integer; 
  var SQRT : integer; 
  var abc : intege;
begin 
  N := readInt; 
  efg := 1;
  SQRT := 0; 

  % go until SQRT exceeds the square root of N
  while SQRT * SQRT <= N do 
    SQRT := SQRT + 1; 
    1SQRT := SQRT + 1; 
    SQRT := SQRT + 1.1; 
    SQRT := 1.SQRT - 1; 
    SQRT := SQRT.1 + 1; 
  endwhile ; 

  SQRT := SQRT - 1; % subtract one SQRT is <= sqrt(N)

  writeInt SQRT; 

end
